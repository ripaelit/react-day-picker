## Getting the selected day

Use `onSelect` to get the selected day

```jsx showOutput open=no
function Example() {
  const [selectedDay, setSelectedDay] = useState();
  return (
    <>
      <DayPicker onSelect={setSelectedDay} />
      <p>
        Please pick a day.
        {selectedDay && <>You selected {selectedDay.toLocaleDateString()}.</>}
      </p>
    </>
  );
}
```

## Pre-selecting a day

Use `defautSelected` to initialize DayPicker with a pre-selected day. In combination with
`required`, DayPicker will always have a day selected.

```jsx showOutput open=no
function Example() {
  const defaultSelected = new Date();
  const [selectedDay, setSelectedDay] = useState(defaultSelected);
  return (
    <>
      <DayPicker
        defaultSelected={defaultSelected}
        required
        onSelect={setSelectedDay}
      />
      <p>
        Please pick a day.
        {selectedDay && <>You selected {selectedDay.toLocaleDateString()}.</>}
      </p>
    </>
  );
}
```

## Selecting multiple days

Using `type="multiple"` to allow the selection of multiple days:

```jsx showOutput open=no
function Example() {
  // Store the selected days in an array
  const defaultSelected = [new Date()];
  const [days, setDays] = useState(defaultSelected);

  const handleResetClick = () => setSelectedDay();

  return (
    <>
      <DayPicker type="multiple" onSelect={setDays} required />
      <p>You picked {days.map((day) => day.toLocaleDateString()).join(', ')}</p>
    </>
  );
}
```

## Selecting a range of days

Using `type="range"` to allow the selection of multiple days. Pass a
[DateRange](/api/types/daterange) object.

```jsx showOutput open=no
function Example() {
  const defaultSelected = {
    from: new Date(),
    to: addDays(new Date(), 4)
  };
  const [range, setRange] = useState(defaultSelected);

  return (
    <>
      <DayPicker
        defaultSelected={defaultSelected}
        type="range"
        onSelect={setRange}
      />
      <p>
        You picked from {range && range.from && range.from.toLocaleDateString()}{' '}
        to {range && range.to && range.to.toLocaleDateString()}
      </p>
    </>
  );
}
```
