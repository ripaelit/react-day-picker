```jsx preview
import React, { useState } from 'react';
import { DayPicker } from 'react-day-picker';
import { format } from 'date-fns';

function App() {
  // Add a "pin" emoji when the date is the 12th or the 17h of the month
  const formatDay = day => {
    const isPinned = (day.getDate() === 12 ||  day.getDate() === 17);
    const pin = isPinned ? 'ðŸ“Œ' : '';
    return `${day.getDate()}${pin}`;
  };

  const formatCaption = (month, { locale }) => {
    return `${format(month, 'dd/yyyy', { locale })}`;
  };

  const formatWeekdayName = (day, { locale }) => {
    return format(day, 'E', { locale })[0];
  };

  return (
    <DayPicker
      showOutsideDays
      showWeekNumber
      formatDay={formatDay}
      formatCaption={formatCaption}
      formatWeekdayName={formatWeekdayName}
    />
  );
}
```
